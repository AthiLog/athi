var inputElems = document.getElementsByTagName("input");
var total = 0;
var count = 0;
var result = 0;
for (var i = 0; i < inputElems.length; i++) {
  if (inputElems[i].type === "checkbox") {
    total++;
  }
  if (inputElems[i].checked === true) {

  }
}
